	bouton_presse (l_event: POINTER; l_barre: FORME_OBJET)
			-- effectue les déplacements appropriés de la barre selon l'event envoyé en paramètre
		local
			l_keyboard_event, l_keysym: POINTER
		do
			l_keyboard_event := {SDL_WRAPPER}.get_sdl_eventkey (l_event)
			l_keysym := {SDL_WRAPPER}.get_sdl_keysym (l_keyboard_event)
			if {SDL_WRAPPER}.get_sdl_sym (l_keysym) = {SDL_WRAPPER}.sdlk_down then
				if l_barre.y < 500.to_integer_16 then
					l_barre.set_player1_y (10.to_integer_16)
				end
			end
			if {SDL_WRAPPER}.get_sdl_sym (l_keysym) = {SDL_WRAPPER}.sdlk_up then
				if l_barre.y > 20.to_integer_16 then
					l_barre.set_player1_y (-10.to_integer_16)
				end
			end
		end

	balle (l_balle: BALLE; l_barre1: FORME_OBJET; l_barre2: FORME_OBJET)
			-- déplace la balle selon ses variables (monte, droite, vitesse)
		local
			l_colision_32: INTEGER_32
		do
			if l_balle.droite then
				if l_balle.x + l_balle.vitesse > 1220.to_integer_16 then
					if l_balle.y >= l_barre2.y + 50.to_integer_16 and l_balle.y + 30.to_integer_16 <= l_barre2.y + 150.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (1220 - (l_balle.vitesse.to_integer_32 - (1220 - l_balle.x.to_integer_32)))
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (False)
					end
					if l_balle.y + 30.to_integer_16 >= l_barre2.y and l_balle.y < l_barre2.y + 50.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (1220 - (l_balle.vitesse.to_integer_32 - (1220 - l_balle.x.to_integer_32)))
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (False)
						l_balle.set_monte (True)
					end
					if l_balle.y + 30.to_integer_16 > l_barre2.y + 150.to_integer_16 and l_balle.y <= l_barre2.y + 200.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (1220 - (l_balle.vitesse.to_integer_32 - (1220 - l_balle.x.to_integer_32)))
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (False)
						l_balle.set_monte (False)
					end
				end
			end
			if l_balle.droite = False then
				if l_balle.x - l_balle.vitesse < 30.to_integer_16 then
					if l_balle.y >= l_barre1.y + 50.to_integer_16 and l_balle.y + 30.to_integer_16 <= l_barre1.y + 150.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (30 + (l_balle.vitesse - (l_balle.x - 30.to_integer_16)).to_integer_32)
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (True)
					end
					if l_balle.y + 30.to_integer_16 >= l_barre1.y and l_balle.y < l_barre1.y + 50.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (30 + (l_balle.vitesse - (l_balle.x - 30.to_integer_16)).to_integer_32)
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (True)
						l_balle.set_monte (True)
					end
					if l_balle.y + 30.to_integer_16 > l_barre1.y + 150.to_integer_16 and l_balle.y <= l_barre1.y + 200.to_integer_16 then
						l_colision_32 := l_balle.x.to_integer_32 - (30 + (l_balle.vitesse - (l_balle.x - 30.to_integer_16)).to_integer_32)
						l_balle.set_x (l_colision_32.as_integer_16)
						l_balle.set_droite (True)
						l_balle.set_monte (False)
					end
				end
			end
			if l_balle.monte then
				if l_balle.y - l_balle.vitesse < 30.to_integer_16 then
					l_balle.set_monte (False)
				end
			end
			if l_balle.monte = False then
				if l_balle.y + l_balle.vitesse > 670.to_integer_16 then
					l_balle.set_monte (True)
				end
			end
			if l_balle.droite then
				l_balle.set_x (l_balle.vitesse)
			end
			if l_balle.droite = False then
				l_balle.set_x (- l_balle.vitesse)
			end
			if l_balle.monte then
				l_balle.set_player1_y (- l_balle.vitesse)
			end
			if l_balle.monte = False then
				l_balle.set_player1_y (l_balle.vitesse)
			end
		end
